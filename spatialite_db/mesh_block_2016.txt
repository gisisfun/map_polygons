

DROP TABLE IF EXISTS "MB_2016";

CREATE TABLE MB_2016 AS
SELECT *, ST_Collect(geometry) FROM
(SELECT *,geometry from MB_2016_ACT
UNION ALL
SELECT *,geometry from MB_2016_NSW
UNION ALL
SELECT *,geometry from MB_2016_VIC
UNION ALL
SELECT *,geometry from MB_2016_TAS
UNION ALL
SELECT *,geometry from MB_2016_SA
UNION ALL
SELECT *,geometry from MB_2016_NT
UNION ALL
SELECT *,geometry from MB_2016_OT
UNION ALL
SELECT *,geometry from MB_2016_WA
UNION ALL
SELECT *,geometry from MB_2016_QLD
);



.quit
